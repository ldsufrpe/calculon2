\begin{align*} \frac{\partial }{\partial x} \left( \frac{\partial f}{\partial x} \right) \amp = \frac{\partial^2 f}{\partial x^2} = f_{xx} \amp \quad\ct{derive duas em relação a x.}\\ \frac{\partial }{\partial y} \left( \frac{\partial f}{\partial y} \right) \amp = \frac{\partial^2 f}{\partial y^2} = f_{yy} \amp \quad\ct{derive duas em relação a y.}\\ \frac{\partial }{\partial y} \left( \frac{\partial f}{\partial x} \right) \amp = \frac{\partial^2 f}{\partial y\partial x} = f_{xy} \amp \quad\ct{derive primeiro relação a x e depois a y.}\\ \frac{\partial }{\partial x} \left( \frac{\partial f}{\partial y} \right) \amp = \frac{\partial^2 f}{\partial x\partial y} = f_{yx} \amp \quad\ct{derive primeiro relação a y e depois a x.} \end{align*}
